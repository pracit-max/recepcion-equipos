function doPost(e){
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);
    
    const data = JSON.parse(e.postData.contents);
    
    // === VALIDACIÓN DE DISPONIBILIDAD (CORREGIDA Y CON LOCK) ===
    if (data.equipo && data.equipo !== "Otros equipos" && data.sede && data.hora_devolucion) {
      const sheet = SpreadsheetApp.openById(SHEET_ID).getSheets()[0];
      const registros = sheet.getDataRange().getValues();
      registros.shift(); 

      const fechaHoy = new Date().toISOString().split('T')[0];
      const horaActualStr = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "HH:mm");

      for (let row of registros) {
        if (row.length < 24) continue; // Ahora 24 columnas

        let fechaRegistro = row[0];
        if (fechaRegistro instanceof Date) {
          fechaRegistro = Utilities.formatDate(fechaRegistro, Session.getScriptTimeZone(), "yyyy-MM-dd");
        } else {
          fechaRegistro = String(fechaRegistro).split('T')[0].split(' ')[0];
        }

        const sedeRegistrada = String(row[4] || "").trim();        // Columna E (índice 4)
        const equipoRegistrado = String(row[5] || "").trim();      // Columna F (índice 5)
        const horaDevolucionStr = String(row[23] || "16:00").trim(); // Columna X (índice 23)

        if (fechaRegistro === fechaHoy && 
            equipoRegistrado === data.equipo && 
            sedeRegistrada.toLowerCase() === data.sede.toLowerCase()) {

          if (horaActualStr < horaDevolucionStr) {

    // === Si pasó la validación → Guardar el registro ===
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheets()[0];
    
    const fotoDanoURL = data.foto_dano ? guardarImagen(data.foto_dano,"dano",FOLDER_DANO) : "";

function guardarImagen(base64,nombre,folderId){

  const folder = DriveApp.getFolderById(folderId);

  const base64Data = base64.includes(",") ? base64.split(",")[1] : base64;
  const decoded = Utilities.base64Decode(base64Data);

  const blob = Utilities.newBlob(decoded,"image/png",nombre+"_"+Date.now()+".png");

  const file = folder.createFile(blob);

  return file.getUrl();

}

/**
 * Obtiene una imagen desde Google Drive y la convierte a base64 (data URI)
 * @param {string} urlOrId - URL de Drive o ID del archivo
 * @returns {string|null} - Data URI de la imagen o null si falla
 */
function obtenerImagenBase64(urlOrId) {
  try {
    // Extraer ID de la URL si es necesario
    let fileId = urlOrId;
    if (urlOrId.includes('drive.google.com')) {
      const match = urlOrId.match(/[-\w]{25,}/);
      if (match) fileId = match[0];
    }
    
    const file = DriveApp.getFileById(fileId);
    const blob = file.getBlob();
    const base64 = Utilities.base64Encode(blob.getBytes());
    const mimeType = blob.getContentType();
    return `data:${mimeType};base64,${base64}`;
  } catch (e) {
    console.error("Error obteniendo imagen: " + e.toString());
    return null;
  }
}

      return ContentService
        .createTextOutput(JSON.stringify(equipos))
        .setMimeType(ContentService.MimeType.JSON);

    } catch(err) {
      console.error(err);
      return ContentService
        .createTextOutput(JSON.stringify({error: err.toString()}))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  // === CASO 3: Traer registros guardados (para la vista de registros y PDF) ===
  // Este es el caso por defecto cuando no hay action específico
  try {
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheets()[0];
    const data = sheet.getDataRange().getValues();
    
    if (data.length <= 1) {
      return ContentService
        .createTextOutput(JSON.stringify([]))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    data.shift(); // Quitar headers

    const registros = data.map((row, index) => ({
      id: index + 1,
      fecha: row[0],                 // Fecha
      nombre: row[1],                // Nombre
      cedula: row[2],                // Cédula
      correo: row[3],                // Correo
      equipo: row[4],                // Carro
      cantidad: row[5],              // Cantidad
      cargador: row[6],              // Cargador
      novedad: row[7],               // Novedad
      descripcion: row[8],           // Descripción
      foto_dano: row[9],             // URL Foto daño
      solicita_cambio: row[10],      // Cambio
      serial_cambio: row[11],        // Serial cambio
      foto_cambio: row[12],          // URL Foto cambio
      equipos_adicionales: row[13],  // Adicionales
      cant_adicional: row[14],       // Cant adicional
      serial_adicional: row[15],     // Serial adicional
      observacion: row[16],          // Observación
      firma: row[17],                // URL Firma
      detalle_equipos: row[18] || "", // Lista de seriales/placas
      es_otros_equipos: row[19] || "No",    // NUEVO
      rango_inicio: row[20] || "",          // NUEVO
      rango_fin: row[21] || "",             // NUEVO
      hora_devolucion: row[22] || "16:00"   // NUEVO
    }));

    return ContentService
      .createTextOutput(JSON.stringify(registros))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({error: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

 * @param {string} code - Código de 6 dígitos
 * @returns {boolean} - true si se envió correctamente
 */
function enviarCodigoVerificacion(email, code) {
  try {
    const asunto = "Código de verificación - Sistema de Equipos";
    
    const cuerpo = `
      <h2>Hola,</h2>
      <p>Tu código de verificación para el sistema de recepción de equipos es:</p>
      <div style="background: #f4f4f4; padding: 20px; text-align: center; font-size: 32px; 
                  font-weight: bold; letter-spacing: 10px; margin: 20px 0; 
                  border-radius: 8px; border: 2px solid #4ECDC4;">
        ${code}
      </div>
      <p>Este código expira en 10 minutos.</p>
      <p style="color: #666; font-size: 12px;">Si no solicitaste este código, ignora este mensaje.</p>
    `;
    
    GmailApp.sendEmail(email, asunto, "Tu código es: " + code, {
      htmlBody: cuerpo,
      name: "Sistema de Equipos Innova"
    });
    
    Logger.log("Código enviado a: " + email);
    return true;
    
  } catch (error) {

    return false;
  }

function getCarrosDisponibles() {

  const sheet = SpreadsheetApp.openById(SHEET_ID).getSheets()[0];
  const data = sheet.getDataRange().getValues();
  data.shift();

  const ahora = new Date();
  const ocupados = [];

  data.forEach(row => {
    const equipo = row[4];
    const horaDevolucion = row[22]; // ✔️ corregido

    if (!horaDevolucion) return;

    const [hora, minuto] = horaDevolucion.toString().split(":");
    const fecha = new Date();
    fecha.setHours(hora, minuto);

    if (ahora < fecha) {
      ocupados.push(equipo);
    }
  });

  return ocupados;
}
